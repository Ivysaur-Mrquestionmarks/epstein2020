		}
		hidden_effect = {

			}
			if = {
				limit = { has_completed_focus = NPA_path_of_the_lone_wolf }
				country_event = { id = maga.8 days = 14 }
			}
			IDA = {
				country_event = {
					id = idaho_rework.261
					days = 3
				}
			}
			news_event = { id = maga.3 days = 1 } 
			every_owned_state = { 
				limit = { is_core_of = USA }
				remove_core_of = USA
			}
			if = {
				limit = {
					has_idea = NPA_Motivation_High
				}
				remove_ideas = NPA_Motivation_High
			}
			if = {
				limit = {
					has_idea = NPA_Motivation_Medium
				}
				remove_ideas = NPA_Motivation_Medium
			}
			if = {
				limit = {
					has_idea = NPA_Motivation_Low
				}
				remove_ideas = NPA_Motivation_Low
			}
			if = {
				limit = {
					has_idea = NPA_Motivation_Soldified
				}
				remove_ideas = NPA_Motivation_Soldified
			}
		}
	}
}

	}
	fire_only_once = yes
	#Based DeSantis
	option = {
		name = maga.9.a
		ai_chance = {
			factor = 40
		}
		set_country_flag = vice_president_desantis
		remove_ideas = Generic_Vacant_Vice_President
		add_ideas = USA_ron_desantis_advisor

			limit = {
				FLO = {
					has_completed_focus = FLO_desantis
				}
			}

			factor = 30
		}
